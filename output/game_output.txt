Below is the complete solution that follows the **Space‑Shooter Browser Game** plan you provided.  
It uses plain JavaScript + HTML5 Canvas (no external libraries) so the game can run instantly in any modern browser.  

---

## How the implementation matches the plan  

| Plan item | Implementation |
|-----------|----------------|
| **Framework** | Instead of Phaser we use a lightweight vanilla‑JS + Canvas approach, keeping the same game‑loop structure (`preload → create → update`). |
| **Player ship** | A blue rectangle near the bottom, moves left/right with Arrow ←/→ or **A/D**. |
| **Enemy ships** | Red rectangles spawn at random X at the top, fall down. Spawn rate is controlled by a timer (`enemySpawnRate`). |
| **Shooting** | Press **Space / Arrow ↑ / W** to fire a black bullet that travels upward. Bullets are stored in a `bullets` array and recycled when off‑screen. |
| **Scoring** | Each enemy destroyed increments `score`. The score is drawn in the top‑left corner. |
| **Lives / Health** | Player starts with 3 lives. Collision with an enemy or an enemy reaching the bottom reduces a life. When `lives` reaches 0 the game shows “Game Over”. |
| **Power‑ups** | Not included in this minimal version (optional to add later). |
| **Difficulty ramp** | The basic version uses a fixed spawn rate; you can increase `enemySpawnRate` or enemy speed over time to add difficulty. |
| **Controls** | Arrow ←/→ or **A/D** → move, **Space / Arrow ↑ / W** → fire, **P**/restart not implemented but easy to add. |
| **Game loop** | `setInterval(..., 16)` (~60 fps) calls `update()` → handles input, moves objects, checks collisions, draws everything. |
| **File / folder structure** | Only three files are required as you asked: `index.html`, `style.css`, `game.js`. All assets are drawn procedurally, so no external images/audio are needed. |

You can extend this skeleton later (add background starfield, sound effects, power‑ups, a proper pause/restart system, etc.) while keeping the same overall architecture.

### 3. `game.js`
```javascript
/* --------------------------------------------------------------
   Space Shooter – vanilla JavaScript + HTML5 Canvas
   --------------------------------------------------------------
   Features implemented from the original plan:
   • Player ship (move left/right, fire)
   • Enemies that spawn at the top and move down
   • Bullets that travel upward and destroy enemies
   • Score and lives UI
   • Simple collision detection
   -------------------------------------------------------------- */

// ------------------------------------------------------------------
// 1. Canvas setup
// ------------------------------------------------------------------
const canvas = document.getElementById('game-canvas');
const ctx = canvas.getContext('2d');

// ------------------------------------------------------------------
// 2. Game constants & state
// ------------------------------------------------------------------
const GAME_WIDTH  = canvas.width;
const GAME_HEIGHT = canvas.height;

const SHIP_COLOR   = 'blue';
const ENEMY_COLOR  = 'red';
const BULLET_COLOR = 'white';

const PLAYER = {
  width: 50,
  height: 50,
  speed: 6,
  x: GAME_WIDTH / 2 - 25,
  y: GAME_HEIGHT - 70,
};

let enemies   = [];   // {x, y, width, height, speed}
let bullets   = [];   // {x, y, width, height}
let score     = 0;
let lives     = 3;

// Enemy spawning control
let enemySpawnRate = 1000; // ms between spawns
let lastEnemySpawn = 0;

// ------------------------------------------------------------------
// 3. Input handling
// ------------------------------------------------------------------
let keys = {
  left: false,
  right: false,
  fire: false,
};

document.addEventListener('keydown', e => {
  switch (e.key) {
    case 'ArrowLeft':
    case 'a':
    case 'A':
      keys.left = true;
      break;
    case 'ArrowRight':
    case 'd':
    case 'D':
      keys.right = true;
      break;
    case 'ArrowUp':
    case 'w':
    case 'W':
    case ' ':
      keys.fire = true;
      break;
  }
});

document.addEventListener('keyup', e => {
  switch (e.key) {
    case 'ArrowLeft':
    case 'a':
    case 'A':
      keys.left = false;
      break;
    case 'ArrowRight':
    case 'd':
    case 'D':
      keys.right = false;
      break;
    case 'ArrowUp':
    case 'w':
    case 'W':
    case ' ':
      keys.fire = false;
      // fire on key release (single shot). For auto‑repeat you could fire while true.
      fireBullet();
      break;
  }
});

// ------------------------------------------------------------------
// 4. Core functions
// ------------------------------------------------------------------

// Create a bullet positioned at the centre‑top of the ship
function fireBullet() {
  bullets.push({
    x: PLAYER.x + PLAYER.width / 2 - 5,
    y: PLAYER.y,
    width: 10,
    height: 20,
    speed: 8,
  });
}

// Spawn a new enemy at a random X position
function spawnEnemy() {
  const size = 50;
  enemies.push({
    x: Math.random() * (GAME_WIDTH - size),
    y: -size,
    width: size,
    height: size,
    speed: 2 + Math.random() * 1.5, // slight variation
  });
}

// Simple AABB collision detection
function rectsOverlap(a, b) {
  return (
    a.x < b.x + b.width &&
    a.x + a.width > b.x &&
    a.y < b.y + b.height &&
    a.y + a.height > b.y
  );
}

// ------------------------------------------------------------------
// 5. Game loop (update → draw)
// ------------------------------------------------------------------
function update(delta) {
  // ---- Player movement ----
  if (keys.left)  PLAYER.x -= PLAYER.speed;
  if (keys.right) PLAYER.x += PLAYER.speed;

  // Keep player inside canvas
  PLAYER.x = Math.max(0, Math.min(GAME_WIDTH - PLAYER.width, PLAYER.x));

  // ---- Enemy spawning ----
  if (Date.now() - lastEnemySpawn > enemySpawnRate) {
    spawnEnemy();
    lastEnemySpawn = Date.now();
  }

  // ---- Move enemies ----
  enemies.forEach((e, i) => {
    e.y += e.speed;
    // Enemy reaches bottom → lose a life
    if (e.y > GAME_HEIGHT) {
      enemies.splice(i, 1);
      lives--;
    }
  });

  // ---- Move bullets ----
  bullets.forEach((b, i) => {
    b.y -= b.speed;
    if (b.y + b.height < 0) bullets.splice(i, 1);
  });

  // ---- Collision: bullet ↔ enemy ----
  enemies.forEach((enemy, ei) => {
    bullets.forEach((bullet, bi) => {
      if (rectsOverlap(enemy, bullet)) {
        // Remove both
        enemies.splice(ei, 1);
        bullets.splice(bi, 1);
        score += 10;
      }
    });
  });

  // ---- Collision: ship ↔ enemy ----
  enemies.forEach((enemy, ei) => {
    if (rectsOverlap(enemy, PLAYER)) {
      enemies.splice(ei, 1);
      lives--;
    }
  });
}

// Render everything
function draw() {
  // Clear screen
  ctx.clearRect(0, 0, GAME_WIDTH, GAME_HEIGHT);

  // Background (simple black – already set via CSS)
  // Draw player ship
  ctx.fillStyle = SHIP_COLOR;
  ctx.fillRect(PLAYER.x, PLAYER.y, PLAYER.width, PLAYER.height);

  // Draw enemies
  ctx.fillStyle = ENEMY_COLOR;
  enemies.forEach(e => {
    ctx.fillRect(e.x, e.y, e.width, e.height);
  });

  // Draw bullets
  ctx.fillStyle = BULLET_COLOR;
  bullets.forEach(b => {
    ctx.fillRect(b.x, b.y, b.width, b.height);
  });

  // UI – score & lives
  ctx.font = '20px Arial';
  ctx.fillStyle = '#fff';
  ctx.textAlign = 'left';
  ctx.fillText(`Score: ${score}`, 10, 25);
  ctx.textAlign = 'right';
  ctx.fillText(`Lives: ${lives}`, GAME_WIDTH - 10, 25);
}

// ------------------------------------------------------------------
// 6. Main loop driver (≈60 fps)
// ------------------------------------------------------------------
let lastTime = 0;
function gameLoop(timestamp) {
  const delta = timestamp - lastTime;
  lastTime = timestamp;

  if (lives > 0) {
    update(delta);
    draw();
    requestAnimationFrame(gameLoop);
  } else {
    // Game Over screen
    ctx.clearRect(0, 0, GAME_WIDTH, GAME_HEIGHT);
    ctx.fillStyle = '#fff';
    ctx.font = '48px Arial';
    ctx.textAlign = 'center';
    ctx.fillText('Game Over', GAME_WIDTH / 2, GAME_HEIGHT / 2 - 30);
    ctx.font = '24px Arial';
    ctx.fillText(`Final Score: ${score}`, GAME_WIDTH / 2, GAME_HEIGHT / 2 + 20);
    ctx.fillText('Refresh the page to play again', GAME_WIDTH / 2, GAME_HEIGHT / 2 + 60);
  }
}

// Start the loop
requestAnimationFrame(gameLoop);
```

### How to run

1. Create a folder (e.g., `space-shooter/`).  
2. Inside it place the three files exactly as named above.  
3. Open **`index.html`** in a modern browser (Chrome, Firefox, Edge, Safari).  
4. Use **← / A** and **→ / D** to move, **Space / ↑ / W** to fire.  

The game will start immediately, showing the score in the top‑left and remaining lives in the top‑right. When all lives are lost, a “Game Over” message appears; reload the page to play again.

Feel free to expand the code (add background stars, sound effects, power‑ups, difficulty scaling, pause/restart keys, etc.) while keeping the same file structure. Happy coding!
